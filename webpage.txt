<script language="vbscript" type="text/vbscript">  
  Sub Sample ()  
   Dim iRow  
   Set objExcel = CreateObject ("Excel. Application")  
   Set objWorkbook = objExcel.Workbooks.Open ("C:\Users\Karthikeyan.K\Desktop\Book1.xlsx")  
   objExcel.Application.Visible = True  
   ObjWorkbook.Windows (1).Visible = True  
   Set XlSheet =objWorkbook.Sheets (1)  
   XlSheet.Activate  
   iRow = 1  
   With objExcel  
       Do while .Cells (iRow, 1).value <> ""  
           .Cells (iRow, 1).activate  
           iRow = iRow + 1  
       Loop  
          .Cells (iRow, 1).value=Document.GetElementsByName ("name") (0).Value  
          .Cells (iRow, 2).value=Document.GetElementsByName ("gender") (0).Value  
          .Cells (iRow, 3).value=Document.GetElementsByName ("date_of_birth") (0).Value  
          .Cells (iRow, 4).value=Document.GetElementsByName ("email") (0).Value  
          .Cells (iRow, 5).value=Document.GetElementsByName ("password") (0).Value
          .Cells (iRow, 6).value=Document.GetElementsByName ("confirm_password") (0).Value
          MsgBox "Data Added Successfully‚Äù, vbinformation  
          Document.GetElementsByName ("name") (0).Value=""  
          Document.GetElementsByName ("gender") (0).Value=""  
          Document.GetElementsByName ("date_of_birth") (0).Value=""  
          Document.GetElementsByName ("email") (0).Value=""  
          Document.GetElementsByName ("password") (0).Value=""  
          Document.GetElementsByName ("confirm_password") (0).Value=""  
      End With  
      ObjWorkbook. Save  
      ObjWorkbook. Close  
      Set objWorkbook = Nothing  
      Set objExcel = Nothing  
  End Sub  
</script>
